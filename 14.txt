2 — Simulation harness

Folder: tools/sim/highdsimharness.ts

`ts
// tools/sim/highdsimharness.ts

import { HighDConsoleIntegration } from "../../src/integration/highdconsoleintegration";
import timelineData from "../../examples/highd_demo/timeline.json";
import capsData from "../../examples/highd_demo/capabilities.json";
import anomsData from "../../examples/highd_demo/anomalies.json";

async function run() {
  const engine = new HighDConsoleIntegration();
  engine.registerAll();

  const timeline = {
    entries: timelineData.entries,
    explanations: {}
  };

  const result = engine.buildAll(timeline, capsData, anomsData);

  console.log("Overlays:", JSON.stringify(result.overlays, null, 2));
  console.log("Stability:", JSON.stringify(result.stability, null, 2));
  console.log("Consistency:", JSON.stringify(result.consistency, null, 2));
  console.log("Signatures:", JSON.stringify(result.signatures.slice(0, 2), null, 2));
}

run().catch(console.error);
`

4 — CLI tool

Folder: tools/cli/highdconsolecli.ts

`ts
// tools/cli/highdconsolecli.ts

import fs from "fs";
import path from "path";
import { HighDConsoleIntegration } from "../../src/integration/highdconsoleintegration";

function loadJSON(p: string) {
  return JSON.parse(fs.readFileSync(p, "utf-8"));
}

async function main() {
  const base = path.resolve(dirname, "../../examples/highd_demo");

  const timelineData = loadJSON(path.join(base, "timeline.json"));
  const capsData = loadJSON(path.join(base, "capabilities.json"));
  const anomsData = loadJSON(path.join(base, "anomalies.json"));

  const engine = new HighDConsoleIntegration();
  engine.registerAll();

  const timeline = {
    entries: timelineData.entries,
    explanations: {}
  };

  const { overlays, stability, consistency } = engine.buildAll(
    timeline,
    capsData,
    anomsData
  );

  console.log("=== High-D Stability Summary ===");
  console.log(JSON.stringify(stability, null, 2));

  console.log("\n=== High-D Consistency Issues ===");
  console.log(JSON.stringify(consistency.issues, null, 2));

  console.log("\n=== High-D Overlay Bands (first 10) ===");
  console.log(JSON.stringify(overlays.bands.slice(0, 10), null, 2));
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
`
