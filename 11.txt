`ts
// plugins/highdvisualizationprimitives/index.ts

import {
  RegimeTimelineEntry,
} from "../../src/runtime/regimetimelineengine";

import {
  AnomalyEvent,
} from "../../src/runtime/anomalyoverlayengine";

export interface Vec3 {
  x: number;
  y: number;
  z: number;
}

export interface HighDVisPrimitive {
  type: string;          // "sphere", "vector", "ribbon", "line", "glyph"
  label: string;
  color: string;
  data: any;             // geometry payload
  timestamp: string;
}

export interface HighDVisFrame {
  timestamp: string;
  primitives: HighDVisPrimitive[];
}

export class HighDVisualizationPrimitives {
  /
   * Build visualization primitives for a single timestamp.
   */
  buildFrame(
    entry: RegimeTimelineEntry,
    caps: Record<string, number>,
    anomalies: AnomalyEvent[]
  ): HighDVisFrame {
    const ts = entry.timestamp;
    const primitives: HighDVisPrimitive[] = [];

    const r4 = caps["hd.r4"] ?? 0;
    const relA = caps["hd.rel3A_mag"] ?? 0;
    const relB = caps["hd.rel3B_mag"] ?? 0;
    const grad = caps["hd.grad_norm"] ?? 0;
    const flux = caps["hd.flux_norm"] ?? 0;
    const tensor = caps["hd.tensor_norm"] ?? 0;
    const anis = caps["cap.hd.anisotropy"] ?? 0;
    const rel = caps["cap.hd.relational_imbalance"] ?? 0;
    const curv = caps["cap.hd.curvature"] ?? 0;

    /* -----------------------------------------------------------
     * 1. Collapse Sphere (4D radius)
      --------------------------------------------------------- /
    primitives.push({
      type: "sphere",
      label: "4D Collapse Radius",
      color: "#6366f1",
      timestamp: ts,
      data: {
        radius: Math.min(1, r4),
      },
    });

    /* -----------------------------------------------------------
     * 2. Relational Subspace Vectors
      --------------------------------------------------------- /
    primitives.push({
      type: "vector",
      label: "Relational Subspace A",
      color: "#10b981",
      timestamp: ts,
      data: {
        magnitude: Math.min(1, relA),
        direction: this.unitDirection(1, 0, 0),
      },
    });

    primitives.push({
      type: "vector",
      label: "Relational Subspace B",
      color: "#ef4444",
      timestamp: ts,
      data: {
        magnitude: Math.min(1, relB),
        direction: this.unitDirection(0, 1, 0),
      },
    });

    /* -----------------------------------------------------------
     * 3. Gradient Arrow
      --------------------------------------------------------- /
    primitives.push({
      type: "vector",
      label: "High-D Gradient",
      color: "#0ea5e9",
      timestamp: ts,
      data: {
        magnitude: Math.min(1, grad),
        direction: this.unitDirection(0, 0, 1),
      },
    });

    /* -----------------------------------------------------------
     * 4. Flux Arrow
      --------------------------------------------------------- /
    primitives.push({
      type: "vector",
      label: "High-D Flux",
      color: "#f59e0b",
      timestamp: ts,
      data: {
        magnitude: Math.min(1, flux),
        direction: this.unitDirection(1, 1, 0),
      },
    });

    /* -----------------------------------------------------------
     * 5. Tensor Glyph (anisotropy)
      --------------------------------------------------------- /
    primitives.push({
      type: "glyph",
      label: "High-D Tensor Glyph",
      color: "#a855f7",
      timestamp: ts,
      data: {
        anisotropy: anis,
        tensorNorm: tensor,
      },
    });

    /* -----------------------------------------------------------
     * 6. Curvature Ribbon
      --------------------------------------------------------- /
    if (curv > 0.5) {
      primitives.push({
        type: "ribbon",
        label: "Curvature Ribbon",
        color: "#7c3aed",
        timestamp: ts,
        data: {
          width: curv,
          intensity: curv,
        },
      });
    }

    /* -----------------------------------------------------------
     * 7. Relational Asymmetry Band
      --------------------------------------------------------- /
    if (rel > 0.5) {
      primitives.push({
        type: "band",
        label: "Relational Asymmetry Band",
        color: "#dc2626",
        timestamp: ts,
        data: {
          thickness: rel,
        },
      });
    }

    /* -----------------------------------------------------------
     * 8. Anisotropy Fracture Line
      --------------------------------------------------------- /
    if (anis > 0.88) {
      primitives.push({
        type: "line",
        label: "Anisotropy Fracture",
        color: "#b91c1c",
        timestamp: ts,
        data: {
          length: anis,
        },
      });
    }

    /* -----------------------------------------------------------
     * 9. High-D Anomaly Markers
      --------------------------------------------------------- /
    for (const a of anomalies) {
      if (!a.id.startsWith("anomaly.hd.")) continue;

      primitives.push({
        type: "marker",
        label: a.label,
        color: this.colorForAnomaly(a.id),
        timestamp: ts,
        data: {
          severity: a.severity,
        },
      });
    }

    return { timestamp: ts, primitives };
  }

  /
   * Build visualization frames for an entire timeline.
   */
  buildTimelineFrames(
    timeline: RegimeTimelineEntry[],
    allCaps: Record<string, Record<string, number>>,
    allAnoms: Record<string, AnomalyEvent[]>
  ): HighDVisFrame[] {
    return timeline.map((entry) =>
      this.buildFrame(
        entry,
        allCaps[entry.timestamp] ?? {},
        allAnoms[entry.timestamp] ?? []
      )
    );
  }

  /* -----------------------------------------------------------
   * Helpers
    --------------------------------------------------------- /
  private unitDirection(x: number, y: number, z: number): Vec3 {
    const mag = Math.sqrt(x  x + y  y + z * z) || 1;
    return { x: x / mag, y: y / mag, z: z / mag };
  }

  private colorForAnomaly(id: string): string {
    if (id.includes("curvature")) return "#7c3aed";
    if (id.includes("flux")) return "#2563eb";
    if (id.includes("gradient")) return "#059669";
    if (id.includes("coupling")) return "#d97706";
    if (id.includes("asymmetry")) return "#dc2626";
    if (id.includes("anisotropy")) return "#b91c1c";
    return "#6b7280";
  }
}
`
