// plugins/highdstabilitydetector/index.ts

import {
  RegimeTimelineEntry,
} from "../../src/runtime/regimetimelineengine";

import {
  AnomalyEvent,
} from "../../src/runtime/anomalyoverlayengine";

export interface HighDStabilityEvent {
  timestamp: string;
  stabilityScore: number;       // 0 = unstable, 1 = stable
  class: "stable" | "warning" | "unstable";
  type: string;                 // e.g. "curvature_bifurcation"
  message: string;
  confidence: number;
}

export interface HighDStabilityReport {
  events: HighDStabilityEvent[];
}

export class HighDStabilityDetector {
  /
   * Compute stability for a single timestamp.
   */
  evaluate(
    entry: RegimeTimelineEntry,
    caps: Record<string, number>,
    anomalies: AnomalyEvent[]
  ): HighDStabilityEvent[] {
    const ts = entry.timestamp;
    const events: HighDStabilityEvent[] = [];

    const r4 = caps["hd.r4"] ?? 0;
    const grad = caps["hd.grad_norm"] ?? 0;
    const flux = caps["hd.flux_norm"] ?? 0;
    const tensor = caps["hd.tensor_norm"] ?? 0;
    const anis = caps["cap.hd.anisotropy"] ?? 0;
    const rel = caps["cap.hd.relational_imbalance"] ?? 0;
    const curv = caps["cap.hd.curvature"] ?? 0;
    const coup = caps["cap.hd.coupling_strength"] ?? 0;

    /* -----------------------------------------------------------
     * 1. Curvature Bifurcation
      --------------------------------------------------------- /
    if (curv > 0.85 && tensor > r4 * 0.8) {
      events.push({
        timestamp: ts,
        stabilityScore: 0.1,
        class: "unstable",
        type: "curvature_bifurcation",
        message: "High-D curvature approaching bifurcation threshold.",
        confidence: curv,
      });
    }

    /* -----------------------------------------------------------
     * 2. Gradient Blow-Up Instability
      --------------------------------------------------------- /
    if (grad > 0.90) {
      events.push({
        timestamp: ts,
        stabilityScore: 0.05,
        class: "unstable",
        type: "gradient_blowup",
        message: "High-D gradients indicate imminent blow-up.",
        confidence: grad,
      });
    }

    /* -----------------------------------------------------------
     * 3. Flux Oscillatory Instability
      --------------------------------------------------------- /
    if (flux > 0.80 && grad > 0.60) {
      events.push({
        timestamp: ts,
        stabilityScore: 0.3,
        class: "warning",
        type: "flux_oscillation",
        message: "High-D flux and gradients suggest oscillatory instability.",
        confidence: (flux + grad) / 2,
      });
    }

    /* -----------------------------------------------------------
     * 4. Coupling Divergence
      --------------------------------------------------------- /
    if (coup > 0.75 && tensor > 0.5) {
      events.push({
        timestamp: ts,
        stabilityScore: 0.2,
        class: "unstable",
        type: "coupling_divergence",
        message: "High-D coupling strength indicates divergence.",
        confidence: coup,
      });
    }

    /* -----------------------------------------------------------
     * 5. Relational Subspace Divergence
      --------------------------------------------------------- /
    if (rel > 0.70) {
      events.push({
        timestamp: ts,
        stabilityScore: 0.4,
        class: "warning",
        type: "relational_divergence",
        message: "Relational subspaces diverging beyond tolerance.",
        confidence: rel,
      });
    }

    /* -----------------------------------------------------------
     * 6. Anisotropy Fracture Instability
      --------------------------------------------------------- /
    if (anis > 0.88) {
      events.push({
        timestamp: ts,
        stabilityScore: 0.05,
        class: "unstable",
        type: "anisotropy_fracture",
        message: "High-D anisotropy indicates structural fracture.",
        confidence: anis,
      });
    }

    /* -----------------------------------------------------------
     * 7. General Stability Score (fallback)
      --------------------------------------------------------- /
    if (events.length === 0) {
      events.push({
        timestamp: ts,
        stabilityScore: 0.95,
        class: "stable",
        type: "stable",
        message: "High-D structure stable.",
        confidence: 0.95,
      });
    }

    return events;
  }

  /
   * Evaluate stability across the entire timeline.
   */
  evaluateTimeline(
    timeline: RegimeTimelineEntry[],
    allCaps: Record<string, Record<string, number>>,
    allAnoms: Record<string, AnomalyEvent[]>
  ): HighDStabilityReport {
    const events: HighDStabilityEvent[] = [];

    for (const entry of timeline) {
      const caps = allCaps[entry.timestamp] ?? {};
      const anoms = allAnoms[entry.timestamp] ?? [];
      events.push(...this.evaluate(entry, caps, anoms));
    }

    return { events };
  }
}
