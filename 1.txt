3. File: plugins/highdcollapsepack/operators.ts

`ts
// plugins/highdcollapsepack/operators.ts

/
 * Generic L2 norm of a vector.
 */
export function l2Norm(values: number[]): number {
  let sum = 0;
  for (const v of values) sum += v * v;
  return Math.sqrt(sum);
}

/
 * L2 norm of a flattened tensor.
 */
export function tensorNorm(flat: number[]): number {
  return l2Norm(flat);
}

/
 * Optional normalization helper.
 */
export function normalize(value: number, scale?: number): number {
  if (!scale || scale === 0) return value;
  return value / scale;
}
`

4. File: plugins/highdcollapsepack/index.ts

`ts
// plugins/highdcollapsepack/index.ts

import {
  Measurement,
  MeasurementPlugin,
} from "../../src/core/types";
import { l2Norm, tensorNorm, normalize } from "./operators";

interface HighDCollapseConfig {
  dimensionGroups: {
    base4: string[];   // e.g. ["coord.x", "coord.y", "coord.z", "coord.w"]
    rel3A?: string[];  // e.g. ["coord.v", "coord.u", "coord.q"]
    rel3B?: string[];  // e.g. ["coord.p", "coord.s", "coord.f"]
  };
  collapseRules: {
    r4: string;
    rel3A_mag?: string;
    rel3B_mag?: string;
    grad_norm?: string;
    flux_norm?: string;
    tensor_norm?: string;
  };
  normalization?: {
    r4_scale?: number;
    rel3A_scale?: number;
    rel3B_scale?: number;
    grad_scale?: number;
    flux_scale?: number;
    tensor_scale?: number;
  };
}

/
 * Helper to read a scalar measurement.
 */
function m(measurements: Record<string, unknown>, id: string): number {
  const v = measurements[id];
  if (typeof v !== "number") {
    throw new Error(Expected measurement "${id}");
  }
  return v;
}

/
 * Helper to read a vector measurement (array of numbers).
 */
function mv(measurements: Record<string, unknown>, id: string): number[] {
  const v = measurements[id];
  if (!Array.isArray(v)) {
    throw new Error(Expected vector measurement "${id}");
  }
  return v.map((x) => {
    if (typeof x !== "number") {
      throw new Error(Expected numeric component in "${id}");
    }
    return x;
  });
}

/
 * Build high-D collapse measurements from a config.
 */
export function buildHighDCollapseMeasurements(
  config: HighDCollapseConfig
): Measurement[] {
  const { dimensionGroups, collapseRules, normalization } = config;

  const measurements: Measurement[] = [];

  // 1. 4D radius: base4 -> r4
  measurements.push({
    id: collapseRules.r4,
    label: "4D Radius",
    description: "L2 norm of the base 4D coordinate group.",
    dependencies: dimensionGroups.base4,
    evaluate: (meas) => {
      const vals = dimensionGroups.base4.map((id) => m(meas, id));
      const r4 = l2Norm(vals);
      return normalize(r4, normalization?.r4_scale);
    },
  });

  // 2. Relational 3D magnitude A
  if (dimensionGroups.rel3A && collapseRules.rel3A_mag) {
    measurements.push({
      id: collapseRules.rel3A_mag,
      label: "Relational 3D Magnitude A",
      description: "L2 norm of the first relational 3D group.",
      dependencies: dimensionGroups.rel3A,
      evaluate: (meas) => {
        const vals = dimensionGroups.rel3A!.map((id) => m(meas, id));
        const mag = l2Norm(vals);
        return normalize(mag, normalization?.rel3A_scale);
      },
    });
  }

  // 3. Relational 3D magnitude B
  if (dimensionGroups.rel3B && collapseRules.rel3B_mag) {
    measurements.push({
      id: collapseRules.rel3B_mag,
      label: "Relational 3D Magnitude B",
      description: "L2 norm of the second relational 3D group.",
      dependencies: dimensionGroups.rel3B,
      evaluate: (meas) => {
        const vals = dimensionGroups.rel3B!.map((id) => m(meas, id));
        const mag = l2Norm(vals);
        return normalize(mag, normalization?.rel3B_scale);
      },
    });
  }

  // 4. High-D gradient norm (expects a vector measurement)
  if (collapseRules.grad_norm) {
    measurements.push({
      id: collapseRules.grad_norm,
      label: "High-D Gradient Norm",
      description: "L2 norm of a high-dimensional gradient vector.",
      dependencies: ["hd.grad_vector"],
      evaluate: (meas) => {
        const grad = mv(meas, "hd.grad_vector");
        const g = l2Norm(grad);
        return normalize(g, normalization?.grad_scale);
      },
    });
  }

  // 5. High-D flux norm (expects a vector measurement)
  if (collapseRules.flux_norm) {
    measurements.push({
      id: collapseRules.flux_norm,
      label: "High-D Flux Norm",
      description: "L2 norm of a high-dimensional flux vector.",
      dependencies: ["hd.flux_vector"],
      evaluate: (meas) => {
        const flux = mv(meas, "hd.flux_vector");
        const f = l2Norm(flux);
        return normalize(f, normalization?.flux_scale);
      },
    });
  }

  // 6. High-D tensor norm (expects a flattened tensor vector)
  if (collapseRules.tensor_norm) {
    measurements.push({
      id: collapseRules.tensor_norm,
      label: "High-D Tensor Norm",
      description: "L2 norm of a flattened high-dimensional tensor.",
      dependencies: ["hd.tensor_flat"],
      evaluate: (meas) => {
        const flat = mv(meas, "hd.tensor_flat");
        const t = tensorNorm(flat);
        return normalize(t, normalization?.tensor_scale);
      },
    });
  }

  return measurements;
}

/
 * Default plugin instance with a canonical config.
 * You can also build custom instances by calling buildHighDCollapseMeasurements
 * with your own config and registering them in the measurement registry.
 */
const defaultConfig: HighDCollapseConfig = {
  dimensionGroups: {
    base4: ["coord.x", "coord.y", "coord.z", "coord.w"],
    rel3A: ["coord.v", "coord.u", "coord.q"],
    rel3B: ["coord.p", "coord.s", "coord.f"],
  },
  collapseRules: {
    r4: "hd.r4",
    rel3Amag: "hd.rel3Amag",
    rel3Bmag: "hd.rel3Bmag",
    gradnorm: "hd.gradnorm",
    fluxnorm: "hd.fluxnorm",
    tensornorm: "hd.tensornorm",
  },
  normalization: {
    r4_scale: 1.0,
    rel3A_scale: 1.0,
    rel3B_scale: 1.0,
    grad_scale: 1.0,
    flux_scale: 1.0,
    tensor_scale: 1.0,
  },
};

export const HighDCollapsePlugin: MeasurementPlugin = {
  namespace: "highdcollapsepack",
  measurements: buildHighDCollapseMeasurements(defaultConfig),
};
`
