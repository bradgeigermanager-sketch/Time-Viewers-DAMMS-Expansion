Capability API

Overview

The capability engine converts collapse metrics into normalized high‑D capability scalars in \([0,1]\).

Capability IDs

- cap.hd.curvature
- cap.hd.flux_intensity
- cap.hd.gradient_intensity
- cap.hd.coupling_strength
- cap.hd.relational_imbalance
- cap.hd.anisotropy

TypeScript

`ts
type CapabilityVector = Record<string, number>;
`

Usage

`ts
import { CapabilityRegistry } from "../../src/core/registry";

const caps: CapabilityVector = CapabilityRegistry.evaluateAll(measurements);
console.log(caps["cap.hd.curvature"]);
`
`

3 — Interactive demo UI

A small React app that wires the sandbox‑like interaction into the browser.

File: src/ui/demo/HighDDemoApp.tsx

`tsx
import React, { useState } from "react";
import { HighDConsoleIntegration } from "../../integration/highdconsoleintegration";

const engine = new HighDConsoleIntegration();
engine.registerAll();

export const HighDDemoApp: React.FC = () => {
  const [inputs, setInputs] = useState({
    r4: "0.6",
    rel3A_mag: "0.4",
    rel3B_mag: "0.2",
    grad_norm: "0.8",
    flux_norm: "0.7",
    tensor_norm: "0.9",
  });

  const [output, setOutput] = useState<any | null>(null);

  const onChange = (k: string, v: string) =>
    setInputs((s) => ({ ...s, [k]: v }));

  const run = () => {
    const caps = {
      t0: {
        "hd.r4": parseFloat(inputs.r4) || 0,
        "hd.rel3Amag": parseFloat(inputs.rel3Amag) || 0,
        "hd.rel3Bmag": parseFloat(inputs.rel3Bmag) || 0,
        "hd.gradnorm": parseFloat(inputs.gradnorm) || 0,
        "hd.fluxnorm": parseFloat(inputs.fluxnorm) || 0,
        "hd.tensornorm": parseFloat(inputs.tensornorm) || 0,
      },
    };

    const timeline = {
      entries: [{ timestamp: "t0", stateId: "s0", regimes: [], meta: {} }],
      explanations: {},
    };

    const anoms: Record<string, any[]> = { t0: [] };

    const result = engine.buildAll(timeline, caps, anoms);
    setOutput(result);
  };

  return (
    <div style={{ padding: "1.5rem", fontFamily: "system-ui" }}>
      <h1>High‑D Demo</h1>
      <p>Adjust collapse metrics and compute regimes, stability, and visualization primitives.</p>

      <div style={{ display: "grid", gap: "0.5rem", maxWidth: 400 }}>
        {Object.keys(inputs).map((k) => (
          <label key={k}>
            {k}:{" "}
            <input
              type="number"
              step="0.01"
              min="0"
              max="1"
              value={inputs[k as keyof typeof inputs]}
              onChange={(e) => onChange(k, e.target.value)}
            />
          </label>
        ))}
      </div>

      <button style={{ marginTop: "1rem" }} onClick={run}>
        Run
      </button>

      {output && (
        <div style={{ marginTop: "1.5rem", display: "grid", gap: "1rem" }}>
          <section>
            <h2>Stability</h2>
            <pre>{JSON.stringify(output.stability.events, null, 2)}</pre>
          </section>
          <section>
            <h2>Consistency Issues</h2>
            <pre>{JSON.stringify(output.consistency.issues, null, 2)}</pre>
          </section>
          <section>
            <h2>Visualization (t0)</h2>
            <pre>{JSON.stringify(output.vis[0], null, 2)}</pre>
          </section>
        </div>
      )}
    </div>
  );
};
`

You can mount this in a simple SPA entry:

`tsx

4 — Large‑scale synthetic dataset generator

Generates many timelines with random high‑D behaviors.

File: tools/data/highdsyntheticgenerator.ts

`ts
// tools/data/highdsyntheticgenerator.ts

import fs from "fs";
import path from "path";

interface SyntheticConfig {
  timelines: number;
  steps: number;
}

function rand(min: number, max: number) {
  return min + Math.random() * (max - min);
}

function genCaps(step: number, steps: number) {
  const phase = step / steps;

  // simple pattern: calm → unstable → calm
  const spike = Math.exp(-Math.pow((phase - 0.5) * 6, 2));

  return {
    "hd.r4": rand(0.4, 0.9),
    "hd.rel3A_mag": rand(0.1, 0.9),
    "hd.rel3B_mag": rand(0.1, 0.9),
    "hd.grad_norm": 0.2 + 0.8 * spike,
    "hd.flux_norm": 0.2 + 0.8 * spike,
    "hd.tensor_norm": 0.2 + 0.8 * spike,
    "cap.hd.curvature": 0.2 + 0.8 * spike,
    "cap.hd.flux_intensity": 0.2 + 0.8 * spike,
    "cap.hd.gradient_intensity": 0.2 + 0.8 * spike,
    "cap.hd.coupling_strength": rand(0.1, 1.0),
    "cap.hd.relational_imbalance": rand(0.0, 1.0),
    "cap.hd.anisotropy": rand(0.0, 1.0),
  };
}

function generate(config: SyntheticConfig) {
  const outDir = path.resolve(dirname, "../../examples/highd_synth");
  fs.mkdirSync(outDir, { recursive: true });

  for (let i = 0; i < config.timelines; i++) {
    const timelineId = tl_${i};
    const entries: any[] = [];
    const caps: Record<string, any> = {};
    const anoms: Record<string, any[]> = {};

    for (let s = 0; s < config.steps; s++) {
      const ts = t${s};
      entries.push({ timestamp: ts, stateId: ${timelineId}_${ts}, regimes: [], meta: {} });
      caps[ts] = genCaps(s, config.steps);
      anoms[ts] = []; // anomalies can be derived later by the engine
    }

    fs.writeFileSync(
      path.join(outDir, ${timelineId}_timeline.json),
      JSON.stringify({ entries }, null, 2)
    );
    fs.writeFileSync(
      path.join(outDir, ${timelineId}_capabilities.json),
      JSON.stringify(caps, null, 2)
    );
    fs.writeFileSync(
      path.join(outDir, ${timelineId}_anomalies.json),
      JSON.stringify(anoms, null, 2)
    );
  }

  console.log("Generated synthetic datasets in", outDir);
}

generate({ timelines: 20, steps: 100 });
`

You can wire this into package.json:

`json
"scripts": {
  "data:gen": "ts-node tools/data/highdsyntheticgenerator.ts"
}
`
