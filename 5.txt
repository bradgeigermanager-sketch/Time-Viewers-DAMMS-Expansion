// plugins/highdanomalypack/index.ts

import {
  Anomaly,
  AnomalyPlugin,
  AnomalyDetectionResult,
} from "../../src/core/types";

/
 * Helper to read a capability.
 */
function c(caps: Record<string, unknown>, id: string): number {
  const v = caps[id];
  if (typeof v !== "number") {
    throw new Error(Expected capability "${id}");
  }
  return v;
}

/
 * Threshold helper.
 */
function above(x: number, t: number): boolean {
  return x >= t;
}

/
 * High‑D anomaly definitions:

 * - flux surge
 * - gradient blow-up
 * - coupling instability

 * - anisotropy fracture

const highDAnomalies: Anomaly[] = [
