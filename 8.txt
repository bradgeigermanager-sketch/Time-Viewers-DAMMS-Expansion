📦 File: plugins/highdsignatureexport_pack/index.ts

`ts
// plugins/highdsignatureexport_pack/index.ts

import {
  RegimeTimelineEntry,
  RegimeTimeline,
} from "../../src/runtime/regimetimelineengine";

import {
  AnomalyEvent,
} from "../../src/runtime/anomalyoverlayengine";

import {
  RegimeExplanation,
} from "../../src/core/types";

export interface HighDSignature {
  timestamp: string;
  stateId?: string;

  collapse: {
    r4?: number;
    rel3A_mag?: number;
    rel3B_mag?: number;
    grad_norm?: number;
    flux_norm?: number;
    tensor_norm?: number;
  };

  capabilities: Record<string, number>;

  regimes: Array<{
    regimeId: string;
    label: string;
    confidence: number;
  }>;

  anomalies: Array<{
    anomalyId: string;
    label: string;
    severity: string;
    score: number;
  }>;

  structuralEnvelope?: {
    dominantRegime: string;
    confidence: number;
  };
}

export class HighDSignatureExportEngine {
  /
   * Build a high‑D signature for a single timestamp.
   */
  buildSignature(
    entry: RegimeTimelineEntry,
    capabilities: Record<string, number>,
    anomalies: AnomalyEvent[],
    explanation?: RegimeExplanation
  ): HighDSignature {
    const collapse = {
      r4: capabilities["hd.r4"],
      rel3Amag: capabilities["hd.rel3Amag"],
      rel3Bmag: capabilities["hd.rel3Bmag"],
      gradnorm: capabilities["hd.gradnorm"],
      fluxnorm: capabilities["hd.fluxnorm"],
      tensornorm: capabilities["hd.tensornorm"],
    };

    const regimes = entry.regimes
      .filter((r) => r.regimeId.startsWith("regime.hd."))
      .map((r) => ({
        regimeId: r.regimeId,
        label: r.label,
        confidence: r.confidence,
      }));

    const anomalyList = anomalies
      .filter((a) => a.id.startsWith("anomaly.hd."))
      .map((a) => ({
        anomalyId: a.id,
        label: a.label,
        severity: a.severity,
        score: a.meta?.score ?? 1,
      }));

    const structuralEnvelope = explanation
      ? {
          dominantRegime: explanation.dominantRegime,
          confidence: explanation.confidence,
        }
      : undefined;

    return {
      timestamp: entry.timestamp,
      stateId: entry.stateId,
      collapse,
      capabilities,
      regimes,
      anomalies: anomalyList,
      structuralEnvelope,
    };
  }

  /
   * Build signatures for an entire timeline.
   */
  buildTimelineSignatures(
    timeline: RegimeTimeline,
    allCapabilities: Record<string, Record<string, number>>,
    allAnomalies: Record<string, AnomalyEvent[]>
  ): HighDSignature[] {
    const signatures: HighDSignature[] = [];

    for (const entry of timeline.entries) {
      const caps = allCapabilities[entry.timestamp] ?? {};
      const anoms = allAnomalies[entry.timestamp] ?? [];
      const exp = timeline.explanations[entry.timestamp];

      signatures.push(
        this.buildSignature(entry, caps, anoms, exp)
      );
    }

    return signatures;
  }
}
`
