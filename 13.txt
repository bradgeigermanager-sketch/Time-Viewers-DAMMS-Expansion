import { HighDCollapsePlugin } from "../../plugins/highdcollapsepack";
import { MeasurementRegistry } from "../../src/core/registry";

describe("High-D Collapse Pack", () => {
  beforeAll(() => {
    MeasurementRegistry.registerPlugin(HighDCollapsePlugin);
  });

  test("computes r4 correctly", () => {
    const meas = {
      "coord.x": 3,
      "coord.y": 4,
      "coord.z": 0,
      "coord.w": 0,
    };

    const r4 = MeasurementRegistry.evaluate("hd.r4", meas);
    expect(r4).toBeCloseTo(5);
  });

  test("relational magnitudes never exceed r4", () => {
    const meas = {
      "coord.x": 10,
      "coord.y": 0,
      "coord.z": 0,
      "coord.w": 0,
      "coord.v": 7,
      "coord.u": 0,
      "coord.q": 0,
    };

    const r4 = MeasurementRegistry.evaluate("hd.r4", meas);
    const relA = MeasurementRegistry.evaluate("hd.rel3A_mag", meas);

    expect(relA).toBeLessThanOrEqual(r4);
  });
});
`

import { HighDConsoleIntegration } from "../../src/integration/highdconsoleintegration";

export class HighDSDK {
  private engine = new HighDConsoleIntegration();

  constructor() {
    this.engine.registerAll();
  }

  evaluate(timeline, caps, anoms) {
    return this.engine.buildAll(timeline, caps, anoms);
  }

  visualize(timeline, caps, anoms) {
    return this.engine.vis.buildTimelineFrames(
      timeline.entries,
      caps,
      anoms
    );
  }

  exportSignatures(timeline, caps, anoms) {
    return this.engine.signatures.buildTimelineSignatures(
      timeline,
      caps,
      anoms
    );
  }
}
`

import json
import subprocess
from pathlib import Path

class HighDSDK:
    def init(self, node_path="node"):
        self.node = node_path
        self.script = Path(file).parent / "bridge.js"

    def evaluate(self, timeline, caps, anoms):
        payload = {
            "timeline": timeline,
            "capabilities": caps,
            "anomalies": anoms
        }
        result = subprocess.check_output(
            [self.node, str(self.script)],
            input=json.dumps(payload).encode("utf-8")
        )
        return json.loads(result.decode("utf-8"))
`
